document.getElementById("signupForm").addEventListener("submit", function(event) { event.preventDefault(); // Prevent the default form submission // Get form data const username = document.getElementById("username").value; const email = document.getElementById("email").value; const password = document.getElementById("password").value; // Create a payload with the form data const data = new URLSearchParams(); data.append("username", username); data.append("email", email); data.append("password", password); // Send data to the backend using fetch API fetch("https://backend-tyb5.onrender.com/signup", { method: "POST", mode: "cors", // Ensure that CORS is enabled for communication headers: { "Content-Type": "application/x-www-form-urlencoded" }, body: data }) .then(response => { if (response.ok) { // Check if the response status is OK return response.json(); // Parse the JSON response } else { return Promise.reject("Signup failed"); // If the request failed, reject with an error } }) .then(data => { // If signup is successful alert(data.message || "Signup successful!"); // Optionally, redirect to another page window.location.href = "/login"; // Change this if you need to redirect elsewhere }) .catch(error => { console.error("Error:", error); alert("An error occurred. Please try again."); }); });
